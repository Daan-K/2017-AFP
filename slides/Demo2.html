data Either (a b : Set) : Set where
  Inl : a -> Either a b
  Inr : b -> Either a b

data Regular : Set where
  Id : Regular
  _|+|_ : Regular -> Regular -> Regular
  _|*|_ : Regular -> Regular -> Regular
  K : U -> Regular

elRegular : Regular -> Set -> Set
elRegular Id x = x
elRegular (r |+| r₁) x = Either (elRegular r x) (elRegular r₁ x)
elRegular (r |*| r₁) x = Pair (elRegular r x) (elRegular r₁ x)
elRegular (K x) x₁ = el x

data Fix (r : Regular) : Set where
  In : elRegular r (Fix r) -> Fix r

[_] : Regular -> Set -> Set
[ r ] = elRegular r

map : (r : Regular) {a b : Set} -> (a -> b) -> [ r ] a -> [ r ] b
map Id f x = f x
map (r |+| r₁) f (Inl x) = Inl (map r f x)
map (r |+| r₁) f (Inr x) = Inr (map r₁ f x)
map (r |*| r₁) f (x , x₁) = (map r f x) , (map r₁ f x₁)
map (K u) f x = x

foldNat Zero zero succ = zero
foldNat (Succ x) zero succ = succ (foldNat x zero succ)

foldList : {a b : Set} -> b -> (a -> b -> b) -> List a -> b
foldList nil cons Nil = nil
foldList nil cons (Cons x xs) = cons x (foldList nil cons xs)

elimList : {a : Set} {P : List a -> Set} ->
  P Nil -> ((x : a) -> (xs : List a) -> P xs -> P (Cons x xs)) ->
  (xs : List a) -> P xs
elimList nil cons Nil = nil
elimList nil cons (Cons x xs) = cons x xs (elimList nil cons xs)

elimVec : {a : Set} (P : {n : ℕ} -> Vec a n -> Set) ->
  P Nil ->
  ((n : ℕ) (x : a) (xs : Vec a n) -> P xs -> P (Cons x xs)) -> 
  (n : ℕ) -> (xs : Vec a n) -> P xs
elimVec P nil cons .0 Nil = nil
elimVec P nil cons _ (Cons x xs) = cons _ x xs (elimVec P nil cons _ xs)
  

double : ℕ -> ℕ
double Zero = Zero
double (Succ n) = Succ (Succ (double n))

data Parity : ℕ -> Set where
  IsOdd : (k : ℕ) -> Parity (Succ (double k)) -- n is 2 * k + 1
  IsEven : (k : ℕ) -> Parity (double k) -- n  is 2 * k

parity? : (n : ℕ) -> Parity n
parity? Zero = IsEven Zero
parity? (Succ Zero) = IsOdd Zero
parity? (Succ (Succ n)) with parity? n
parity? (Succ (Succ .(Succ (double k)))) | IsOdd k = IsOdd (Succ k)
parity? (Succ (Succ .(double k))) | IsEven k = IsEven (Succ k)

useParity : {P : ℕ -> Set} -> (n : ℕ) -> P n
useParity n with parity? n
useParity .(Succ (double k)) | IsOdd k = {!!}
useParity .(double k) | IsEven k = {!!}

snoc : {a : Set} -> (x : a) -> (xs : List a) -> List a
snoc x Nil = Cons x Nil
snoc x (Cons x₁ xs) = Cons x₁ (snoc x xs)

data SnocView {a : Set} : List a -> Set where
  Nil : SnocView Nil
  Snoc : (x : a) -> (xs : List a) -> SnocView (snoc x xs)

snocview? : {a : Set} -> (xs : List a) -> SnocView xs
snocview? Nil = Nil
snocview? (Cons x xs) with snocview? xs
snocview? (Cons x .Nil) | Nil = Snoc x Nil
snocview? (Cons x₁ .(snoc x xs)) | Snoc x xs = Snoc x (Cons x₁ xs)
