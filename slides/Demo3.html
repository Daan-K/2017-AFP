data Stream (a : Set) : Set where
  Cons : a -> ∞ (Stream a) -> Stream a

headS : {a : Set} -> Stream a -> a
headS (Cons x xs) = x


unfoldStream : {a s : Set} -> (s -> Pair a s) -> s -> Stream a
unfoldStream step seed = let xy = step seed in Cons (fst xy) (♯ (unfoldStream step (snd xy)))

zeros : Stream ℕ

mapStream : ∀ {a b : Set} -> (a -> b) -> Stream a -> Stream b
mapStream f (Cons x xs) = Cons (f x) (♯ (mapStream f (♭ xs)))


data Partial (a : Set) : Set where
  Done : a -> Partial a
  Step : ∞ (Partial a) -> Partial a

petrol : ∀ {a} -> ℕ -> Partial a -> Maybe a
petrol Zero p = Nothing
petrol (Succ n) (Done x) = Just x
petrol (Succ n) (Step x) = petrol n (♭ x)

return : {a : Set} -> a -> Partial a
return x = Done x

_>>=_ : {a b : Set} -> Partial a -> (a -> Partial b) -> Partial b
Done x >>= f = f x
Step x >>= f = Step (♯ (♭ x >>= f))

never : {a : Set} -> Partial a
never = Step (♯ never) 

race : {a : Set} -> Partial a -> Partial a -> Partial a
race (Done x) q = Done x
race (Step x) (Done x₁) = Done x₁
race (Step x) (Step x₁) = Step (♯ race (♭ x) (♭ x₁))

--- Lambda calculus ---

data U : Set where
  i : U
  _=>_ : U -> U -> U

Ctx = List U

data Ref (s : U) : Ctx -> Set where
  Top : ∀ {ctx} -> Ref s (Cons s ctx)
  Pop : ∀ {ctx t} -> Ref s ctx -> Ref s (Cons t ctx)

data Term : Ctx -> U -> Set where
  App : ∀ {Γ t s} -> Term Γ (s => t) -> Term Γ s -> Term Γ t
  Lam : ∀ {Γ t s} -> Term (Cons s Γ) t -> Term Γ (s => t)
  Var : ∀ {Γ s} -> Ref s Γ -> Term Γ s

Val : U -> Set
Val i = Bool
Val (u => u₁) = Val u -> Val u₁

data Env : Ctx -> Set where
  Nil : Env Nil
  Cons : forall {u ctx} -> Val u -> Env ctx -> Env (Cons u ctx)

envLookup : forall {ctx u} -> Ref u ctx -> Env ctx -> Val u
envLookup Top (Cons x ctx₁) = x
envLookup (Pop ref) (Cons x ctx₁) = envLookup ref ctx₁

eval : ∀ {Γ s} -> Term Γ s -> Env Γ -> Val s
eval (App f x) env = (eval f env) (eval x env) 
eval (Lam t) env = \x -> eval t (Cons x env)
eval (Var x) env = envLookup x env
